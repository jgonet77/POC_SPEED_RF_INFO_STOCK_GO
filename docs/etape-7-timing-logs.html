1. Mesurer le Temps Écoulé

fun checkApiHealth() { _apiHealthStatus.value = "Testing..." _apiTestTime.value = "Testing..." // Démarrer le chrono val startTime = System.currentTimeMillis() repository.checkApiHealth( onSuccess = { response -> // Calculer le temps écoulé val elapsed = System.currentTimeMillis() - startTime _apiHealthStatus.value = "✅ ${response.status}" _apiTestTime.value = "Completed in ${elapsed}ms" _errorMessage.value = "" }, onError = { error -> val elapsed = System.currentTimeMillis() - startTime _apiHealthStatus.value = "❌ Failed" _apiTestTime.value = "Failed after ${elapsed}ms" _errorMessage.value = error } ) }

Temps Typiques Observés

Latence réseau: 50-500ms
Parsing JSON: <1ms
Requête SQL: 1-10ms
Total: 50-510ms (surtout network)

2. AppLogger : Journalisation Centralisée

object AppLogger { private val logs = mutableListOf<String>() fun log(message: String) { val timestamp = SimpleDateFormat( "HH:mm:ss.SSS", Locale.getDefault() ).format(Date()) val logEntry = "[$timestamp] $message" logs.add(logEntry) Log.d("AppLogger", logEntry) } fun logApiCall(endpoint: String, method: String) { log("API CALL: $method $endpoint") } fun logApiResponse( endpoint: String, code: Int, body: String ) { log("API RESPONSE: $endpoint [$code] " + "$body") } fun logApiError(endpoint: String, error: String) { log("API ERROR: $endpoint - $error") } fun getLogs(): List<String> = logs.toList() fun clearLogs() { logs.clear() } }
